Zentrale Join-Kette für silver.fact_email:

FROM       tbl_analytics_link a
LEFT JOIN  tbl_email_receiver_status c  ON a.EmailReceiverStatusId = c.Id
LEFT JOIN  tbl_email b                  ON a.EmailId               = b.Id
LEFT JOIN  tbl_hr_employee hr           ON a.TNumber               = hr.T_NUMBER

WHERE      a.IsActive = 1

SQL-Skizze für gold.fact_cross_channel_lite:

WITH m AS (
  SELECT
    REGEXP_EXTRACT(TrackingId, '^([A-Z0-9]+-[0-9]+)', 1) AS tracking_pack_id,
    REGEXP_EXTRACT(TrackingId, '^([A-Z0-9]+)',       1) AS cluster_id,
    COUNT(*)                                            AS mailing_count,
    MIN(StartsentTime)                                  AS first_send_at,
    AVG(elementcounts)                                  AS avg_elementcounts,
    AVG(uniquecomponents)                               AS avg_components,
    AVG(CountLanguages)                                 AS avg_languages
  FROM imep_gold.tbl_pbi_platform_mailings
  WHERE IsActive = TRUE
  GROUP BY 1, 2
),
e AS (
  SELECT
    REGEXP_EXTRACT(TrackingId, '^([A-Z0-9]+-[0-9]+)', 1) AS tracking_pack_id,
    COUNT(*)                                            AS event_count,
    SUM(registration)                                   AS total_registrations,
    MIN(StartDate)                                      AS first_event_at
  FROM imep_gold.tbl_pbi_platform_events
  WHERE IsActive = TRUE
    AND IsLegacy  = FALSE
    AND EventFrom < '2100-01-01'
  GROUP BY 1
)
SELECT
  COALESCE(m.tracking_pack_id, e.tracking_pack_id) AS tracking_pack_id,
  m.cluster_id, m.mailing_count, m.first_send_at,
  m.avg_elementcounts, m.avg_components, m.avg_languages,
  e.event_count, e.total_registrations, e.first_event_at
FROM m FULL OUTER JOIN e USING (tracking_pack_id);
